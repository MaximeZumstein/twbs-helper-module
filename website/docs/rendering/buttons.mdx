// Link button
$oButton = new \Laminas\Form\Element\Button('Link', [
    'label' => 'Link',
    'variant' => 'primary',
    'tag' => 'a',
]);
$oButton->setAttribute('href', '#');
echo $this->formButton($oButton) . PHP_EOL;

// Submit button
$oButton = new \Laminas\Form\Element\Submit('Button', [
    'label' => 'Button',
    'variant' => 'primary',
]);
echo $this->formButton($oButton) . PHP_EOL;
