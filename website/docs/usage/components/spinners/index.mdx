    'success',
    'danger',
    'warning',
    'info',
    'light',
    'dark',
] as $sVariant) {
    echo $this->spinner(

            'variant' => $sVariant,
            'type'    => 'grow',
            'label'   => 'Loading...',
        ]
    );
}
