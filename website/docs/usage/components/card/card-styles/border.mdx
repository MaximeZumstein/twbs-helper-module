foreach ([

    'success',
    'danger',
    'warning',
    'info',
    'light',
    'dark',
] as $sVariant) {
    echo $this->card(
        [
            'header' => 'Header',
            'title'  => 'Primary card title',
            'text'   => 'Some quick example text to build on the card title and '.'make up the bulk of the card\'s content.',
        ],
        [
            'borderVariant' => $sVariant,
            'bodyVariant'   => $sVariant !== 'light' ? $sVariant : null,
            'class'         => 'mb-3',
        ]
    ).PHP_EOL;
}
