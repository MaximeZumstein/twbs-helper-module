```php

$oFactory = new \Laminas\Form\Factory();

echo $this->formRow(

        [
            'name'       => 'username',
            'type'       => 'text',
            'options'    => [
                'form_group'        => false,
                'input_group_class' => 'mb-3',

            ],
            'attributes' => [
                'placeholder'      => 'Username',

            'name'       => 'recipient_username',

                'input_group_class' => 'mb-3',

    )

echo $this->formRow(
    $oFactory->create(
        [
            'name'       => 'url',
            'type'       => 'text',
            'options'    => [
                'label'             => 'Your vanity URL',
                'form_group'        => false,
                'input_group_class' => 'mb-3',
                'add_on_prepend'    => 'https://example.com/users/',

            ],
        ]
    )
).PHP_EOL;

echo $this->formRow(
    $oFactory->create(
        [
            'name'       => 'amount',
            'type'       => 'text',
            'options'    => [
                'form_group'        => false,
                'input_group_class' => 'mb-3',
                'add_on_prepend'    => '$',
                'add_on_append'     => '.00',

            'attributes' => [    'aria-label' => 'Amount (to the nearest dollar)'    ],

echo $this->formRow(
```
