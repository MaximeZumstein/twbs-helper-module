```php
<?php

foreach ([
    'primary',
    'secondary',
    'success',
    'danger',
    'warning',
    'info',
    'light',
    'dark',
] as $sVariant) {
    echo $this->badge(ucfirst($sVariant), $sVariant).PHP_EOL;
}
```
