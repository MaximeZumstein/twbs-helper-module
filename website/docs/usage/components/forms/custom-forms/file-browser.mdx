<TabItem value="source" label="Source">

```php
<?php

$oFactory = new \Laminas\Form\Factory();

echo $this->formRow(
    $oFactory->create(
        [
            'name'       => 'custom_file',
            'type'       => 'file',
            'options'    => [
                'custom'       => true,
                'custom_label' => 'Choose file',
                'form_group'   => false,
            ],
            'attributes' => ['id' => 'customFile'],
        ]
    )
);

```
