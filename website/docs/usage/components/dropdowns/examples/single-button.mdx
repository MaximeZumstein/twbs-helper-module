```php
<?php

echo $this->formButton(
    [
        'name'       => 'dropdown',
        'options'    => [
            'label'    => 'Dropdown',
            'dropdown' => [
                'Action',
                'Another action',
                'Something else here',
            ],
        ],
        'attributes' => ['id' => 'dropdownMenuButton'],
    ]
);


echo PHP_EOL.'<br/>';

// With <a> elements
echo $this->formButton(
    [
        'name'       => 'dropdown',
        'options'    => [
            'tag'      => 'a',
            'label'    => 'Dropdown',
            'dropdown' => [
                'Action',
                'Another action',
                'Something else here',
            ],
        ],
        'attributes' => ['id' => 'dropdownMenuButton'],
    ]
);

echo PHP_EOL.'<br/>';

// Variations
foreach ([
    'primary',
    'secondary',
    'success',
    'danger',
    'warning',
    'info',
    'light',
    'dark',
] as $sVariant) {
    echo $this->formButton(
        [
            'name'    => 'dropdown',
            'options' => [
                'variant'  => $sVariant,
                'label'    => 'Dropdown',
                'dropdown' => [
                    'attributes' => ['class' => 'btn-group'],
                    'items'      => [
                        'Action',
                        'Another action',
                        'Something else here',
                        '---',
                        'Separated link',
                    ],
                ],
            ],
        ]
    );
}//end foreach
```
